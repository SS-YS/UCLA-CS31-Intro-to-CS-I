#include <iostream>
using namespace std;

int main()
{
    int side;

    cout << "Enter a number: ";
    cin >> side;

    int i = 0;
    while (i < side)
    {
        int j = i;
        while (j >= 0) {
            cout << "#";
            j -= 1;
        }
        cout << "\n";
        i += 1;
    }
}
